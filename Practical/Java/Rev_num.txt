public class Rev_num
{
    public static void main(String []ar)
    {
        int num=Integer.parseInt(ar[0]);
        int rev=0,rem;
        while(num>0)
        {
            rem=num%10;
            rev=rev*10+rem;
            num=num/10;

        }
        System.out.println("Reverse of number is : "+rev);
    }
}
