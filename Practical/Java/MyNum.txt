import java.io.*;
import java.lang.*;
public class MyNum
{
    private static int num;
    public MyNum()
    {
        num=0;
    }
    public MyNum(int a)
    {
        this.num=a;
    }
    boolean isNegative()
    {
        if(num<0)
            return true;
        else
            return false;
    }
    boolean isPositive()
    {
        if(num>=0)
            return true;
        else
            return false;
    }
    boolean isOdd()
    {
        if(num%2!=0)
            return true;
        else
            return false;
    }
    boolean isEven()
    {
        if(num%2==0)
            return true;
        else 
            return false;
    }
    public static void main(String []ar)throws IOException
    {

        MyNum mn1=new MyNum();
        MyNum mn2=new MyNum(num);

        if(mn1.isNegative())
            System.out.println("Number is Neagative");
        if(mn2.isNegative())
            System.out.println("Number is Neagative ");
        if(mn1.isPositive())
            System.out.println("Number is Positive ");
        if(mn2.isPositive())
            System.out.println("Number is Positive");
        if(mn1.isOdd())
            System.out.println("Number is Odd ");
        if(mn2.isOdd())
            System.out.println("Number is Odd ");
        if(mn1.isEven())
            System.out.println("Number is Even");
        if(mn2.isEven())
            System.out.println("Number is Even");

    }
}
