import java.io.*;
class InvalidDateException extends Exception
{}
class Date
{
    
    int day,month,year;
    public void accept(int dd,int mm ,int yy)throws IOException
    {
       day=dd;
       month=mm;
       year=yy;
    }
    public void display()
    {
        System.out.println("Date is---> "+day+"/"+month+"/"+year);
    }
}

        int year=Integer.parseInt(br.readLine());

                {
                    if(day>=1 &&  day<=31)

                    {
